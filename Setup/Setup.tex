%% Header:

\long\def\title#1{\def\title{#1}}
\def\author{}
\long\def\author#1{\def\author{#1}}

\newcommand{\Header}[3][]{%
    \begin{adjustwidth}{-1mm}{-1mm} %% Increasing Box width
        \begin{tcolorbox}[%
            arc=0mm, %% Rounding of the Headerbox
            colframe=\ColourOne, %% Colour of the Headerbox
            colback=\ColourOne, %% Colour of the Headerbox
            boxrule=0pt, % Border Box
            toptitle=0mm, %
            bottomtitle=1mm, %
            right=0mm, %
            left=0mm, %
            fontupper=\color{white}, % Colour of the Text
            after skip = 0.1mm,%
            top= -2mm, %
            bottom=0mm, %
            title= \huge \center \textbf{#2} \par% %% Title
            \large \author{} %% Author
            \center \small Version: \today %% Version
        ]%
        \end{tcolorbox}%
    \end{adjustwidth}
}


%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%% Section:

\renewcommand{\section}[1]{
\begin{adjustwidth}{-1mm}{-1mm} %% Increasing Box width
    \begin{tcolorbox}[%
                        arc=0mm, %% Rounding of the Sectionbox
                        colback=\ColourOne, %% Colour of the Sectionbox
                        boxrule=0pt, %% Border Box
                        valign=center, %% Vertically Center Title
                        height from=0mm to 8cm, %% Min and Max Height
                        top=-4pt, %% Padding Top
                        bottom=-4pt, %% Padding Bottom
                        left=0mm,  %% Padding Left
                        right=0mm, %% Padding Right
                        after skip = 0.01mm, %% Skip After
                        before skip = 0.01mm, %% Skip Before
                        fontupper=\color{white} %% Colour of the Text
                        ]%
                        \addtocounter{section}{1} %% Increasing the Section Count
                        \setcounter{subsection}{0}  %% Reset Subsection Counter
                        \setcounter{subsubsection}{0}  %% Reset Subsubsection Counter
                        \renewcommand{\tabcolsep}{6pt} %% Reset Columnseperation
                        \vskip-2pt
                        {{\textbf{
                        \hskip-8pt
                        \begin{tabularx}{\textwidth}{lX}
                            \thesection & #1
                        \end{tabularx}}}}% %% Section Title
    \end{tcolorbox}%
\end{adjustwidth}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%% Subsection:


\renewcommand{\subsection}[1]{
\begin{adjustwidth}{-1mm}{-1mm} %% Increasing Box width
    \begin{tcolorbox}[%
                        arc=0mm, %% Rounding of the Subsectionbox
                        colback=\ColourOne !55, %% Colour of the Subsectionbox
                        boxrule=0pt, %% Border Box
                        valign=center, %% Vertically Center Title
                        height from=0mm to 8cm, %% Min and Max Height
                        top=-4pt, %% Padding Top
                        bottom=-4pt, %% Padding Bottom
                        left=0mm,  %% Padding Left
                        right=0mm, %% Padding Right
                        after skip = 0.01mm, %% Skip After
                        before skip = 0.01mm, %% Skip Before
                        fontupper=\color{black} %% Colour of the Text
                        ]%
                        \addtocounter{subsection}{1} %% Increasing the Subsection Count
                        \setcounter{subsubsection}{0}  %% Reset Subsubsection Counter
                        \renewcommand{\tabcolsep}{6pt} %% Reset Columnseperation
                        \vskip-2pt
                        {{\textbf{
                        \hskip-8pt
                        \begin{tabularx}{\textwidth}{lX}
                            \thesubsection & #1
                        \end{tabularx}}}}% %% Subsection Title
    \end{tcolorbox}%
\end{adjustwidth}
}


%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%% Subsubsection:

\renewcommand{\subsubsection}[1]{
\begin{adjustwidth}{-1mm}{-1mm} %% Increasing Box width
    \begin{tcolorbox}[%
                        arc=0mm, %% Rounding of the Subsubsectionbox
                        colback=\ColourOne !30, %% Colour of the Subsubsectionbox
                        boxrule=0pt, %% Border Box
                        valign=center, %% Vertically Center Title
                        height from=0mm to 8cm, %% Min and Max Height
                        top=-4pt, %% Padding Top
                        bottom=-4pt, %% Padding Bottom
                        left=0mm,  %% Padding Left
                        right=0mm, %% Padding Right
                        after skip = 0.05mm, %% Skip After
                        before skip = 0.05mm, %% Skip Before
                        fontupper=\color{black} %% Colour of the Text
                        ]%
                        \addtocounter{subsubsection}{1} %% Increasing the Subsubsection Count
                        \renewcommand{\tabcolsep}{6pt} %% Reset Columnseperation
                        \vskip-2pt
                        {{\textbf{
                        \hskip-8pt
                        \begin{tabularx}{\textwidth}{l X}
                            \thesubsubsection & #1
                        \end{tabularx}}}}% %% Subsubsection Title
    \end{tcolorbox}%
\end{adjustwidth}
}

%% Smallsectionbox:
\newcommand*{\smallsectionbox}[1]{%
    \begin{adjustwidth}{-1mm}{-1mm}
        \begin{tcolorbox}[%
                        arc=0mm,% corner radius of the subsubsection box
                        colback=\ColourOne!30,%
                        boxrule=0pt,% border box
                        valign=center,% vertically center title
                        height=8pt,
                        top=0mm,
                        bottom=0mm,
                        left=0.2mm,%
                        right=0.2mm,%
                        after skip = 0.01mm,%
                        before skip = 0mm,%
                        fontupper=\color{black}% color of the subsubsection box
                        ]%
                        \textbf{#1}%
        \end{tcolorbox}%
    \end{adjustwidth}%
}
